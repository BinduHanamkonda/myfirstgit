class Palindrome {
Scanner input=new Scanner(System.in);
System.out.println("Enter value for n:" +n);
int n=input.nextInt();
int temp=n;
int rem,rev=0;
while(n!=0) {
	rem=n%10;
	rev=rev*10+rem;
	n=n/10;
}if(temp==rev) {
	System.out.println(temp+"is a palindrome");
}
else{
	System.out.println(temp+"is not a palindrome");
}
}
}
