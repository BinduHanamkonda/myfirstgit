public class Polymorphism {
	class A{
		public void run() {
			System.out.println("A class method");
		}
		}
class B extends A{
	public void run() {
		System.out.println("B class method");
	}
}
class Main{
	public  void main(String[] args) {
		A a1=new A();
		a1.run();
		B b1=new B();
		b1.run();
	}
}
}
